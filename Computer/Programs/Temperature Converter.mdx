# Temperature Converter
This program converts temperature between Celsius and Fahrenheit.

<CodeBlock
        language="python"
        filename="temperatureConverter.py"
        code={`
def celsius_to_fahrenheit(celsius):
    return (celsius * 9/5) + 32

def fahrenheit_to_celsius(fahrenheit):
    return (fahrenheit - 32) * 5/9

# Get input from the user
temp = float(input("Enter temperature: "))
unit = input("Enter unit (C or F): ").upper()

if unit == "C":
    fahrenheit = celsius_to_fahrenheit(temp)
    print(f"{temp}째C is equal to {fahrenheit}째F")
elif unit == "F":
    celsius = fahrenheit_to_celsius(temp)
    print(f"{temp}째F is equal to {celsius}째C")
else:
    print("Invalid unit.")
`}
