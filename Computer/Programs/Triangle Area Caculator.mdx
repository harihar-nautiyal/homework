import math

def calculate_triangle_area(base, height):
    if base <= 0 or height <= 0:
        return "Base and height must be positive values."
    area = 0.5 * base * height
    return area  # This was incorrectly indented

def triangle_area_heron(a, b, c):
    if a <= 0 or b <= 0 or c <= 0:  # Added check for positive values
        return "All sides must be positive values."
    if a + b <= c or a + c <= b or b + c <= a:
        return "Invalid triangle sides."
    s = (a + b + c) / 2
    area = math.sqrt(s * (s - a) * (s - b) * (s - c))
    return area

# Let's add some basic error handling for the inputs
