# Password Generator
This program generates a secure password using a combination of letters, numbers, and special characters.

filename="passwordGenerator.py"
code={"def generate_password(length=12):\n    import random\n    import string\n    \n    min_length = 8\n    if length < min_length:\n        length = min_length\n    \n    characters = string.ascii_letters + string.digits + string.punctuation\n    password = ''.join(random.choice(characters) for _ in range(length))\n    return password\n\n# Example usage\npassword_length = int(input(\"Enter desired password length (minimum 8): \"))\ngenerated_password = generate_password(password_length)\nprint(f\"Your generated password is: {generated_password}\")"}
